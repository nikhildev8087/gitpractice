git init   :for initialize a git
git status :checkout the status of a git
git add file_name    :add new file into the git
git add .      :add all files into the git
git commit -m "your message"    :add the message on staged state
git checkout file_name     : revert back the previous changes of a perticular file
git checkout -f         :revert back the all changes
git log          :show the log of a git
